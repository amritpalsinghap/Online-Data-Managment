Professor Table

1:-CheckProfessor
SELECT COUNT(ProfessorID) AS Expr1
FROM     Professor
WHERE  (UserName = @username) AND (Password = @password)

2:-GetProfessorByUserNameAndPassword
SELECT ProfessorID, FirstName, LastName, Email, Phone, Street, Province, City, PostalCode, Password, UserName
FROM     Professor
WHERE  (UserName = @username) AND (Password = @password)
